```openscad
sr = (10 * 0.9144) / 2;
ch = sr * tan(33);
cv = (1/3) * PI * pow(sr, 2) * ch;
tv = (5 * pow(0.9144, 3)) + (1500 * pow(0.3048, 3)) + 100 + (20 * 44 * 0.00454609);
yh = max(0, (tv - cv) / (PI * pow(sr, 2)));
union() {
    cylinder(h=ch, r1=sr, r2=0, $fn=100);
    translate([0, 0, ch]) cylinder(h=yh, r=sr, $fn=100);
}
```
