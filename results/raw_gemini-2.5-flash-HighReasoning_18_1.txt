// Pyramid Construction Plan for Pharoh
// Units: Meters (SI)

// --- STONE PARAMETERS ---
// Each stone is a 50cm side-length cube.
stone_size = 0.5; // 0.5 meters

// --- BUDGET AND DESIGN CALCULATIONS ---
// Total stones available: 150,000
// Stacking rule: Each stone's 4 base corners must sit in the middle of 4 stones below it.
// This results in a stepped pyramid where each successive layer is one stone shorter
// on each side than the layer below it, and is centered above it.
//
// Let S_0 be the number of stones along one side of the base layer.
// The total number of stones for a pyramid with N layers and S_0 stones at the base (where N = S_0 for a single top stone)
// is the sum of squares from 1 to S_0: Sum(k^2 for k=1 to S_0) = S_0 * (S_0 + 1) * (2*S_0 + 1) / 6.
//
// We need to find the largest S_0 such that the total number of stones <= 150,000.
//
// Testing values:
// S_0 = 70 => 70 * 71 * 141 / 6 = 116,945 stones
// S_0 = 75 => 75 * 76 * 151 / 6 = 143,450 stones
// S_0 = 76 => 76 * 77 * 153 / 6 = 149,466 stones
// S_0 = 77 => 77 * 78 * 155 / 6 = 155,255 stones (exceeds budget)
//
// Therefore, the maximum base side length (S_0) is 76 stones, resulting in 76 layers.
base_stones_per_side = 76; 
num_layers = base_stones_per_side; // The number of layers is equal to the base_stones_per_side for a peak of 1 stone.

// --- PYRAMID DIMENSIONS ---
// Base width: 76 stones * 0.5 m/stone = 38 meters
// Height: 76 layers * 0.5 m/layer = 38 meters

// Calculate half-width of the full base layer for accurate centering over the tomb (0,0,0)
base_half_width = (base_stones_per_side * stone_size) / 2; // 38m / 2 = 19m

// --- PYRAMID GENERATION ---
// Iterate through each layer, starting from the base (layer_k = 0) up to the top (layer_k = num_layers - 1).
for (layer_k = [0 : num_layers - 1]) {
    // Determine the number of stones along one side for the current layer.
    // Each layer is one stone smaller on each side than the layer below it.
    current_layer_stones_per_side = base_stones_per_side - layer_k;

    // Calculate the starting X, Y coordinate for the bottom-left stone of the current layer.
    // This ensures each layer is centered over (0,0) and correctly inset by stone_size/2 from the layer below.
    current_layer_start_coord = - (current_layer_stones_per_per_side * stone_size) / 2;
    
    // Calculate the Z coordinate (height) for the current layer.
    layer_z = layer_k * stone_size;

    // Iterate through each stone within the current layer (i for X-axis, j for Y-axis).
    for (i = [0 : current_layer_stones_per_side - 1]) {
        for (j = [0 : current_layer_stones_per_side - 1]) {
            // Calculate the X and Y positions for the bottom-left corner of the current stone.
            stone_x = current_layer_start_coord + i * stone_size;
            stone_y = current_layer_start_coord + j * stone_size;
            
            // Translate to the stone's position and render a cube of the defined stone_size.
            // OpenSCAD's `cube()` by default creates a cube with its origin at (0,0,0).
            translate([stone_x, stone_y, layer_z]) {
                cube(stone_size);
            }
        }
    }
}

// The final output is a union of all individual stone cubes, representing the stepped pyramid.
// This structure precisely includes every step and is accurate to the millimeter as requested.
