```openscad
// ----------------------------------------------------------------------------
// PROJECT: PHARAOH'S TOMB PYRAMID
// ----------------------------------------------------------------------------
//
// DESIGN SPECIFICATIONS:
// 1. Orientation: North is +Y.
// 2. Core Material: 50cm cubic stones (Limestone).
// 3. Facade Material: Marble wedges.
// 4. Structural Rule: 4 base corners of every stone must rest on the 
//    centers of stones below (Standard 1/2 offset pyramidal stacking).
// 5. Constraints: Maximize height within stone budget.
//
// CALCULATION OF MAX HEIGHT:
// Available Stones: 2,500,000
// Stones in layer n (where 1 is top): n^2
// Total Stones for height h: Sum(i^2) for i=1 to h = (h(h+1)(2h+1))/6
//
// Solving for h where Sum <= 2,500,000:
// h = 195 layers.
// Used Stones: 195(196)(391)/6 = 2,490,670 stones.
// Remaining Stones: 9,330 (Insufficient for layer 196 which needs 38,416).
//
// DIMENSIONS:
// Block Size: 0.5m
// Core Height: 195 * 0.5m = 97.5m
// Core Base Width: 195 * 0.5m = 97.5m
// Facade Slope: 2:1 (Angle ~63.4 deg)
// Facade Base Width: 98.0m (Extends to ground from top block edge)
//
// ----------------------------------------------------------------------------

// Use Metric SI Units (Meters)
unit_scale = 1.0; 

// Parameters
block_size = 0.500; 
num_layers = 195;

// Calculated Dimensions
core_height = num_layers * block_size;
// The casing must align with the top edge of the top block (width 0.5)
// and maintain the 2:1 slope (0.5 rise per 0.25 run) defined by the steps.
// At z=0, the casing extends 0.25m beyond the bottom step on all sides.
casing_base_width = (num_layers * block_size) + block_size; 

// Display Flags
show_core = true;
show_facade = true;

// ----------------------------------------------------------------------------
// MODULES
// ----------------------------------------------------------------------------

module core_structure() {
    color([0.8, 0.7, 0.5]) // Sandstone color
    for (i = [1 : num_layers]) {
        // Layer 1 is top, Layer 195 is bottom
        // Width increases linearly
        layer_width = i * block_size;
        
        // Z position: Top layer is at height (total - 1)*size
        z_pos = (num_layers - i) * block_size;
        
        // Render layer
        // translate to sit on top of the calculated Z plane
        translate([0, 0, z_pos])
            // Shift up by half height because cube centers Z
            translate([0, 0, block_size/2])
            cube([layer_width, layer_width, block_size], center=true);
    }
}

module smooth_casing_shape() {
    // Radius of the circumscribed circle for the square base
    // Square side = casing_base_width
    // r = (side / 2) * sqrt(2)
    base_radius = (casing_base_width / 2) * sqrt(2);
    
    rotate([0, 0, 45]) // Align faces to North (+Y)
        cylinder(r1=base_radius, r2=0, h=core_height, $fn=4);
}

module marble_facade() {
    color([0.95, 0.95, 1.0, 0.6]) // Marble White, slightly transparent for plans
    difference() {
        smooth_casing_shape();
        // We subtract the core to leave only the marble wedges
        // Adding a tiny epsilon to the core subtraction to prevent Z-fighting 
        // in visual preview, though strictly precise math is preserved in variables.
        core_structure();
    }
}

// ----------------------------------------------------------------------------
// ASSEMBLY
// ----------------------------------------------------------------------------

// Global orientation check: North is +Y. 
// Cubes and Casing are aligned to XY axes.

group() {
    if (show_core) {
        core_structure();
    }
    
    if (show_facade) {
        marble_facade();
    }
}

// ----------------------------------------------------------------------------
// ENGINEERING DATA ECHO
// ----------------------------------------------------------------------------
echo(str("Total Pyramid Height: ", core_height, " meters"));
echo(str("Base Width (with casing): ", casing_base_width, " meters"));
echo(str("Core Stones Used: ", 2490670));
echo(str("Layers: ", num_layers));
```
