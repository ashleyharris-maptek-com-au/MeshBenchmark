// Pyramid specifications
// Stone size: 50cm (500mm) cube
// Constraint: Each stone's 4 base corners must sit in middle of stones below
// This means stones are offset by 250mm (half stone size) in each layer

stone_size = 500; // mm
half_stone = stone_size / 2;

// Calculate maximum pyramid height given constraints
// Layer 0 (base): n x n stones
// Layer 1: (n-1) x (n-1) stones (offset by half_stone)
// Layer k: (n-k) x (n-k) stones
// Total stones = n^2 + (n-1)^2 + (n-2)^2 + ... + 1^2 = n(n+1)(2n+1)/6

// For 150,000 stones: n(n+1)(2n+1)/6 <= 150000
// Solving: n â‰ˆ 62
// Verify: 62*63*125/6 = 81,375
// n=70: 70*71*141/6 = 117,285
// n=75: 75*76*151/6 = 143,325
// n=76: 76*77*153/6 = 149,226
// n=77: 77*78*155/6 = 155,155 (too many)

base_width = 76;
num_layers = 76;

module pyramid_layer(layer_num, size) {
    stones_per_side = base_width - layer_num;
    if (stones_per_side > 0) {
        offset = layer_num * half_stone;
        z_height = layer_num * stone_size;
        
        translate([offset, offset, z_height]) {
            cube([stones_per_side * stone_size, stones_per_side * stone_size, stone_size]);
        }
    }
}

module stepped_pyramid() {
    for (layer = [0:num_layers-1]) {
        pyramid_layer(layer, base_width - layer);
    }
}

module smooth_pyramid() {
    base = base_width * stone_size;
    height = num_layers * stone_size;
    apex_offset = (num_layers - 1) * half_stone + half_stone;
    
    polyhedron(
        points = [
            // Base corners
            [0, 0, 0],
            [base, 0, 0],
            [base, base, 0],
            [0, base, 0],
            // Apex
            [apex_offset, apex_offset, height]
        ],
        faces = [
            [0, 1, 2, 3],  // base
            [0, 4, 1],      // north face
            [1, 4, 2],      // east face
            [2, 4, 3],      // south face
            [3, 4, 0]       // west face
        ]
    );
}

// Render smooth pyramid (comment out for stepped version)
smooth_pyramid();
