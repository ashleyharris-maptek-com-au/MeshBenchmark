// Pyramid construction for Pharaoh
// Each stone: 50cm (500mm) cube
// Total stones: 600,000
// Constraint: Each stone's 4 corners rest on centers of 4 stones below

// Stone dimension in mm
stone_size = 500;

// Calculate maximum pyramid layers
// Layer n has n^2 stones
// Total stones = sum from 1 to h of n^2 = h(h+1)(2h+1)/6
// Solving: h(h+1)(2h+1)/6 = 600000
// h â‰ˆ 134 layers

// Verify: 134(135)(269)/6 = 599,495 stones (within budget)
height_layers = 134;

// Each layer offset is stone_size/2 = 250mm inward from layer below
// Height per layer is stone_size = 500mm

module pyramid_layer(layer_num) {
    stones_per_side = layer_num;
    offset = (height_layers - layer_num) * stone_size / 2;
    z_height = (height_layers - layer_num) * stone_size;
    
    for (x = [0:stones_per_side-1]) {
        for (y = [0:stones_per_side-1]) {
            translate([
                offset + x * stone_size,
                offset + y * stone_size,
                z_height
            ])
            cube([stone_size, stone_size, stone_size]);
        }
    }
}

module smooth_pyramid() {
    base_width = height_layers * stone_size;
    pyramid_height = height_layers * stone_size;
    
    linear_extrude(height = pyramid_height, scale = 0)
    square([base_width, base_width]);
}

difference() {
    smooth_pyramid();
    
    // Hollow out internal stepped structure volume that won't be visible
    translate([stone_size/2, stone_size/2, stone_size])
    linear_extrude(height = (height_layers-1) * stone_size, scale = (height_layers-1)/height_layers)
    square([(height_layers-1) * stone_size, (height_layers-1) * stone_size]);
}
